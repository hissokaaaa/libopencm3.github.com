61 /** @brief Set the Number of Wait States
62 
63 Used to match the system clock to the FLASH memory access time. See the
64 programming manual for more information on clock speed and voltage ranges. The
65 latency must be changed to the appropriate value <b>before</b> any increase in
66 clock speed, or <b>after</b> any decrease in clock speed. A latency setting of
67 zero only applies if 64-bit mode is not used.
68 
69 @param[in] ws values from @ref flash_latency.
70 */
71 
72 void flash_set_ws(uint32_t ws)
73 {
74  uint32_t reg32;
75 
76  reg32 = FLASH_ACR;
77  reg32 &= ~(1 << 0);
78  reg32 |= ws;
79  FLASH_ACR = reg32;
80 }

153 /** @brief Write a word to eeprom
154  *
155  * @param address assumed to be in the eeprom space, no checking
156  * @param data word to write
157  */
158 void eeprom_program_word(uint32_t address, uint32_t data)
159 {
160  flash_unlock_pecr();
161  /* erase only if needed */
162  FLASH_PECR &= ~FLASH_PECR_FTDW;
163  MMIO32(address) = data;
164  flash_lock_pecr();
165 }

167 /** @brief Write a block of words to eeprom
168  *
169  * Writes a block of words to EEPROM at the requested address, erasing if
170  * necessary, and locking afterwards. Only wordwise writing is safe for
171  * writing any value
172  *
173  * @param[in] address must point to EEPROM space, no checking!
174  * @param[in] data pointer to data to write
175  * @param[in] length_in_words size of of data in WORDS!
176  */
177 void eeprom_program_words(uint32_t address, uint32_t *data, int length_in_words)
178 {
179  int i;
180  flash_unlock_pecr();
181  while (FLASH_SR & FLASH_SR_BSY);
182  /* erase only if needed */
183  FLASH_PECR &= ~FLASH_PECR_FTDW;
184  for (i = 0; i < length_in_words; i++) {
185  MMIO32(address + (i * sizeof(uint32_t))) = *(data+i);
186  while (FLASH_SR & FLASH_SR_BSY);
187  }
188  flash_lock_pecr();
189 }
