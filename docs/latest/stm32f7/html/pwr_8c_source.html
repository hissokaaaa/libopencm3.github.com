41 void pwr_set_vos_scale(enum pwr_vos_scale scale)
42 {

54 void pwr_enable_overdrive(void)
55 {
56  PWR_CR1 |= PWR_CR1_ODEN;
57  while (!(PWR_CSR1 & PWR_CSR1_ODRDY));
58  PWR_CR1 |= PWR_CR1_ODSWEN;
59  while (!(PWR_CSR1 & PWR_CSR1_ODSWRDY));
60 }

62 void pwr_disable_overdrive(void)
63 {
64  PWR_CR1 &= ~(PWR_CR1_ODEN | PWR_CR1_ODSWEN);
65  while (!(PWR_CSR1 & PWR_CSR1_ODSWRDY));
66 }
