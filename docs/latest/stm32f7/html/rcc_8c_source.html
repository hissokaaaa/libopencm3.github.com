295 void rcc_clock_setup_hse_3v3(const struct rcc_clock_scale *clock)
296 {
297  /* Enable internal high-speed oscillator. */
298  rcc_osc_on(RCC_HSI);
299  rcc_wait_for_osc_ready(RCC_HSI);
300 
301  /* Select HSI as SYSCLK source. */
302  rcc_set_sysclk_source(RCC_CFGR_SW_HSI);
303 
304  /* Enable external high-speed oscillator 8MHz. */
305  rcc_osc_on(RCC_HSE);
306  rcc_wait_for_osc_ready(RCC_HSE);
307 
308  rcc_periph_clock_enable(RCC_PWR);
309  pwr_set_vos_scale(clock->vos_scale);
310 
311  if (clock->overdrive) {
312  pwr_enable_overdrive();
313  }
314 
315  /*
316  * Set prescalers for AHB, ADC, ABP1, ABP2.
317  * Do this before touching the PLL (TODO: why?).
318  */
319  rcc_set_hpre(clock->hpre);
320  rcc_set_ppre1(clock->ppre1);
321  rcc_set_ppre2(clock->ppre2);
322 
323  rcc_set_main_pll_hse(clock->pllm, clock->plln,
324  clock->pllp, clock->pllq);
325 
326  /* Enable PLL oscillator and wait for it to stabilize. */
327  rcc_osc_on(RCC_PLL);
328  rcc_wait_for_osc_ready(RCC_PLL);
329 
330  /* Configure flash settings. */
331  flash_set_ws(clock->flash_config);
332 
333  /* Select PLL as SYSCLK source. */
334  rcc_set_sysclk_source(RCC_CFGR_SW_PLL);
335 
336  /* Wait for PLL clock to be selected. */
337  rcc_wait_for_sysclk_status(RCC_PLL);
338 
339  /* Set the peripheral clock frequencies used. */
340  rcc_apb1_frequency = clock->apb1_frequency;
341  rcc_apb2_frequency = clock->apb2_frequency;
342 
343  /* Disable internal high-speed oscillator. */
344  rcc_osc_off(RCC_HSI);
345 }
