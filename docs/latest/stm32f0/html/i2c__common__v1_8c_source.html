44 /** @brief I2C Reset.
45 
46 The I2C peripheral and all its associated configuration registers are placed in
47 the reset condition. The reset is effected via the RCC peripheral reset system.
48 
49 @param[in] i2c Unsigned int32. I2C peripheral identifier @ref i2c_reg_base.
50 */
51 
52 void i2c_reset(uint32_t i2c)
53 {
54  switch (i2c) {
55  case I2C1:
56  rcc_periph_reset_pulse(RST_I2C1);
57  break;
58 #if defined(I2C2_BASE)
59  case I2C2:
60  rcc_periph_reset_pulse(RST_I2C2);
61  break;
62 #endif
63 #if defined(I2C3_BASE)
64  case I2C3:
65  rcc_periph_reset_pulse(RST_I2C3);
66  break;
67 #endif
68 #if defined(I2C4_BASE)
69  case I2C4:
70  rcc_periph_reset_pulse(RST_I2C4);
71  break;
72 #endif
73  default:
74  break;
75  }
76 }

466 static void i2c_write7_v1(uint32_t i2c, int addr, uint8_t *data, size_t n)
467 {
468  while ((I2C_SR2(i2c) & I2C_SR2_BUSY)) {
469  }
470 
471  i2c_send_start(i2c);
472 
473  /* Wait for master mode selected */
474  while (!((I2C_SR1(i2c) & I2C_SR1_SB)
475  & (I2C_SR2(i2c) & (I2C_SR2_MSL | I2C_SR2_BUSY))));
476 
477  i2c_send_7bit_address(i2c, addr, I2C_WRITE);
478 
479  /* Waiting for address is transferred. */
480  while (!(I2C_SR1(i2c) & I2C_SR1_ADDR));
481 
482  /* Clearing ADDR condition sequence. */
483  (void)I2C_SR2(i2c);
484 
485  for (size_t i = 0; i < n; i++) {
486  i2c_send_data(i2c, data[i]);
487  while (!(I2C_SR1(i2c) & (I2C_SR1_BTF)));
488  }
489 }

491 static void i2c_read7_v1(uint32_t i2c, int addr, uint8_t *res, size_t n)
492 {
493  i2c_send_start(i2c);
494  i2c_enable_ack(i2c);
495 
496  /* Wait for master mode selected */
497  while (!((I2C_SR1(i2c) & I2C_SR1_SB)
498  & (I2C_SR2(i2c) & (I2C_SR2_MSL | I2C_SR2_BUSY))));
499 
500  i2c_send_7bit_address(i2c, addr, I2C_READ);
501 
502  /* Waiting for address is transferred. */
503  while (!(I2C_SR1(i2c) & I2C_SR1_ADDR));
504  /* Clearing ADDR condition sequence. */
505  (void)I2C_SR2(i2c);
506 
507  for (size_t i = 0; i < n; ++i) {
508  if (i == n - 1) {
509  i2c_disable_ack(i2c);
510  }
511  while (!(I2C_SR1(i2c) & I2C_SR1_RxNE));
512  res[i] = i2c_get_data(i2c);
513  }
514  i2c_send_stop(i2c);
515 
516  return;
517 }

519 /**
520  * Run a write/read transaction to a given 7bit i2c address
521  * If both write & read are provided, the read will use repeated start.
522  * Both write and read are optional
523  * There are likely still issues with repeated start/stop condtions!
524  * @param i2c peripheral of choice, eg I2C1
525  * @param addr 7 bit i2c device address
526  * @param w buffer of data to write
527  * @param wn length of w
528  * @param r destination buffer to read into
529  * @param rn number of bytes to read (r should be at least this long)
530  */
531 void i2c_transfer7(uint32_t i2c, uint8_t addr, uint8_t *w, size_t wn, uint8_t *r, size_t rn) {
532  if (wn) {
533  i2c_write7_v1(i2c, addr, w, wn);
534  }
535  if (rn) {
536  i2c_read7_v1(i2c, addr, r, rn);
537  } else {
538  i2c_send_stop(i2c);
539  }
540 }

542 /**
543  * Set the i2c communication speed.
544  * @param i2c peripheral, eg I2C1
545  * @param speed one of the listed speed modes @ref i2c_speeds
546  * @param clock_megahz i2c peripheral clock speed in MHz. Usually, rcc_apb1_frequency / 1e6
547  */
548 void i2c_set_speed(uint32_t i2c, enum i2c_speeds speed, uint32_t clock_megahz)
549 {
550  i2c_set_clock_frequency(i2c, clock_megahz);
551  switch(speed) {
552  case i2c_speed_fm_400k:
553  i2c_set_fast_mode(i2c);
554  i2c_set_ccr(i2c, clock_megahz * 5 / 6);
555  i2c_set_trise(i2c, clock_megahz + 1);
556  break;
557  default:
558  /* fall back to standard mode */
559  case i2c_speed_sm_100k:
560  i2c_set_standard_mode(i2c);
561  /* x Mhz / (100kHz * 2) */
562  i2c_set_ccr(i2c, clock_megahz * 5);
563  /* Sm mode, (100kHz) freqMhz + 1 */
564  i2c_set_trise(i2c, clock_megahz + 1);
565  break;
566  }
567 }
