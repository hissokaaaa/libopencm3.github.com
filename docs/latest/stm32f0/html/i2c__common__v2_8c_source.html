28 /** @brief I2C Reset.
29  *
30  * The I2C peripheral and all its associated configuration registers are placed
31  * in the reset condition. The reset is effected via the RCC peripheral reset
32  * system.
33  *
34  * @param[in] i2c Unsigned int32. I2C peripheral identifier @ref i2c_reg_base.
35  */
36 
37 void i2c_reset(uint32_t i2c)
38 {
39  switch (i2c) {
40  case I2C1:
41  rcc_periph_reset_pulse(RST_I2C1);
42  break;
43 #if defined(I2C2_BASE)
44  case I2C2:
45  rcc_periph_reset_pulse(RST_I2C2);
46  break;
47 #endif
48 #if defined(I2C3_BASE)
49  case I2C3:
50  rcc_periph_reset_pulse(RST_I2C3);
51  break;
52 #endif
53 #if defined(I2C4_BASE)
54  case I2C4:
55  rcc_periph_reset_pulse(RST_I2C4);
56  break;
57 #endif
58  default:
59  break;
60  }
61 }

378 /**
379  * Run a write/read transaction to a given 7bit i2c address
380  * If both write & read are provided, the read will use repeated start.
381  * Both write and read are optional
382  * @param i2c peripheral of choice, eg I2C1
383  * @param addr 7 bit i2c device address
384  * @param w buffer of data to write
385  * @param wn length of w
386  * @param r destination buffer to read into
387  * @param rn number of bytes to read (r should be at least this long)
388  */
389 void i2c_transfer7(uint32_t i2c, uint8_t addr, uint8_t *w, size_t wn, uint8_t *r, size_t rn)
390 {
391  /* waiting for busy is unnecessary. read the RM */
392  if (wn) {
393  i2c_set_7bit_address(i2c, addr);
394  i2c_set_write_transfer_dir(i2c);
395  i2c_set_bytes_to_transfer(i2c, wn);
396  if (rn) {
397  i2c_disable_autoend(i2c);
398  } else {
399  i2c_enable_autoend(i2c);
400  }
401  i2c_send_start(i2c);
402 
403  while (wn--) {
404  bool wait = true;
405  while (wait) {
406  if (i2c_transmit_int_status(i2c)) {
407  wait = false;
408  }
409  while (i2c_nack(i2c)); /* FIXME Some error */
410  }
411  i2c_send_data(i2c, *w++);
412  }
413  /* not entirely sure this is really necessary.
414  * RM implies it will stall until it can write out the later bits
415  */
416  if (rn) {
417  while (!i2c_transfer_complete(i2c));
418  }
419  }
420 
421  if (rn) {
422  /* Setting transfer properties */
423  i2c_set_7bit_address(i2c, addr);
424  i2c_set_read_transfer_dir(i2c);
425  i2c_set_bytes_to_transfer(i2c, rn);
426  /* start transfer */
427  i2c_send_start(i2c);
428  /* important to do it afterwards to do a proper repeated start! */
429  i2c_enable_autoend(i2c);
430 
431  for (size_t i = 0; i < rn; i++) {
432  while (i2c_received_data(i2c) == 0);
433  r[i] = i2c_get_data(i2c);
434  }
435  }
436 }

439 /**
440  * Set the i2c communication speed.
441  * NOTE: 1MHz mode not yet implemented!
442  * Min clock speed: 8MHz for FM, 2Mhz for SM,
443  * @param i2c peripheral, eg I2C1
444  * @param speed one of the listed speed modes @ref i2c_speeds
445  * @param clock_megahz i2c peripheral clock speed in MHz. Usually, rcc_apb1_frequency / 1e6
446  */
447 void i2c_set_speed(uint32_t i2c, enum i2c_speeds speed, uint32_t clock_megahz)
448 {
449  int prescaler;
450  switch(speed) {
451  case i2c_speed_fmp_1m:
452  /* FIXME - add support for this mode! */
453  break;
454  case i2c_speed_fm_400k:
455  /* target 8Mhz input, so tpresc = 125ns */
456  prescaler = clock_megahz / 8 - 1;
457  i2c_set_prescaler(i2c, prescaler);
458  i2c_set_scl_low_period(i2c, 10-1); // 1250ns
459  i2c_set_scl_high_period(i2c, 4-1); // 500ns
460  i2c_set_data_hold_time(i2c, 3); // 375ns
461  i2c_set_data_setup_time(i2c, 4-1); // 500ns
462  break;
463  default:
464  /* fall back to standard mode */
465  case i2c_speed_sm_100k:
466  /* target 4Mhz input, so tpresc = 250ns */
467  prescaler = (clock_megahz / 4) - 1;
468  i2c_set_prescaler(i2c, prescaler);
469  i2c_set_scl_low_period(i2c, 20-1); // 5usecs
470  i2c_set_scl_high_period(i2c, 16-1); // 4usecs
471  i2c_set_data_hold_time(i2c, 2); // 0.5usecs
472  i2c_set_data_setup_time(i2c, 5-1); // 1.25usecs
473  break;
474  }
475 }
