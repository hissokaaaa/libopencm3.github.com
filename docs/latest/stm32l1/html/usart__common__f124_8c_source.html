37 /** @brief USART Send a Data Word.
38 
39 @param[in] usart unsigned 32 bit. USART block register address base @ref
40 usart_reg_base
41 @param[in] data unsigned 16 bit.
42 */
43 
44 void usart_send(uint32_t usart, uint16_t data)
45 {
46  /* Send data. */
47  USART_DR(usart) = (data & USART_DR_MASK);
48 }

51 /** @brief USART Read a Received Data Word.
52 
53 If parity is enabled the MSB (bit 7 or 8 depending on the word length) is the
54 parity bit.
55 
56 @param[in] usart unsigned 32 bit. USART block register address base @ref
57 usart_reg_base
58 @returns unsigned 16 bit data word.
59 */
60 
61 uint16_t usart_recv(uint32_t usart)
62 {
63  /* Receive data. */
64  return USART_DR(usart) & USART_DR_MASK;
65 }

67 /*---------------------------------------------------------------------------*/
68 /** @brief USART Wait for Transmit Data Buffer Empty
69 
70 Blocks until the transmit data buffer becomes empty and is ready to accept the
71 next data word.
72 
73 @param[in] usart unsigned 32 bit. USART block register address base @ref
74 usart_reg_base
75 */
76 
77 void usart_wait_send_ready(uint32_t usart)
78 {
79  /* Wait until the data has been transferred into the shift register. */
80  while ((USART_SR(usart) & USART_SR_TXE) == 0);
81 }
82 
83 /*---------------------------------------------------------------------------*/
84 /** @brief USART Wait for Received Data Available
85 
86 Blocks until the receive data buffer holds a valid received data word.
87 
88 @param[in] usart unsigned 32 bit. USART block register address base @ref
89 usart_reg_base
90 */
91 
92 void usart_wait_recv_ready(uint32_t usart)
93 {
94  /* Wait until the data is ready to be received. */
95  while ((USART_SR(usart) & USART_SR_RXNE) == 0);
96 }

99 /** @brief USART Read a Status Flag.

102 usart_reg_base

106 
107 bool usart_get_flag(uint32_t usart, uint32_t flag)
