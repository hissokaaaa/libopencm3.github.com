44 /** Randomizes a number (non-blocking).
45  * Can fail if a clock error or seed error is detected. Consult the Reference
46  * Manual, but "try again", potentially after resetting the peripheral
47  * @param pointer to a uint32_t that will be randomized.
48  * @returns true on success, pointer is only written to on success
49  * @sa rng_get_random_blocking
50  */
51 bool rng_get_random(uint32_t *rand_nr)
52 {
53  /* data ready */
54  if (!(RNG_SR & RNG_SR_DRDY)) {
55  return false;
56  }
57 
58  /* Check for errors */
59  if (RNG_SR & (RNG_SR_CECS | RNG_SR_SECS)) {
60  return false;
61  }
62 
63  *rand_nr = RNG_DR;
64 
65  return true;
66 }

69 /**
70  * Get a random number and block until it works.
71  * Unless you have a clock problem, this should always return "promptly"
72  * If you have a clock problem, you will wait here forever!
73  * @returns a random 32bit number
74  */
75 uint32_t rng_get_random_blocking(void)
76 {
77  uint32_t rv;
78  bool done;
79  do {
80  if (RNG_SR & RNG_SR_SECS) {
81  rng_disable();
82  rng_enable();
83  }
84  done = rng_get_random(&rv);
85  } while (!done);
86 
87  return rv;
88 }
