40 void timer_set_option(uint32_t timer_peripheral, uint32_t option)
41 {
42  if (timer_peripheral == TIM2) {
43  TIM_OR(timer_peripheral) &= ~TIM2_OR_ITR1_RMP_MASK;
44  TIM_OR(timer_peripheral) |= option;
45  } else if (timer_peripheral == TIM5) {
46  TIM_OR(timer_peripheral) &= ~TIM5_OR_TI4_RMP_MASK;
